# HC-SR04 Ultrasonic Module Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 实现基于 MSPM0 PWM 与 Timer Capture 的 HC-SR04 模块，通过 LibXR Topic 发布带状态和 EMA 滤波的距离样本。

**Architecture:** `HC_SR04` 继承 `LibXR::Application`，构造阶段配置 Trigger PWM，`OnMonitor()` 轮询 Capture 原始事件并发布样本。硬件无关的换算、范围判断和滤波放在 `sr04_math.hpp`，由主机测试覆盖；SysConfig 只负责定时器和引脚基础初始化。

**Tech Stack:** C++20、LibXR Application/Topic、LibXR `MSPM0PWM`、TI MSPM0 DriverLib、CMake/Ninja、PowerShell + g++ 主机测试。

```cpp
#pragma once

#include <cstdint>

namespace Module::HC_SR04Logic
{
enum class Status : uint8_t { NO_ECHO = 0, VALID, BELOW_MIN, ABOVE_MAX };
struct Sample
{
  uint32_t pulse_width_us = 0;
  float raw_distance_mm = 0.0F;
  float filtered_distance_mm = 0.0F;
  Status status = Status::NO_ECHO;
  uint32_t sequence = 0;
};
struct FilterState
{
  float filtered_distance_mm = 0.0F;
  bool has_valid_sample = false;
};
inline uint32_t TicksToMicroseconds(uint32_t ticks, uint32_t capture_clock_hz)
{
  const uint64_t scaled = static_cast<uint64_t>(ticks) * 1000000ULL;
  return static_cast<uint32_t>((scaled + capture_clock_hz / 2U) / capture_clock_hz);
}
inline Status ClassifyDistance(float distance_mm, float min_distance_mm,
                               float max_distance_mm)
{
  if (distance_mm < min_distance_mm) return Status::BELOW_MIN;
  if (distance_mm > max_distance_mm) return Status::ABOVE_MAX;
  return Status::VALID;
}
inline Sample ProcessCapture(uint32_t ticks, uint32_t capture_clock_hz,
                             float min_distance_mm, float max_distance_mm,
                             float filter_old_weight, FilterState& state)
{
  Sample sample;
  sample.pulse_width_us = TicksToMicroseconds(ticks, capture_clock_hz);
  sample.raw_distance_mm = static_cast<float>(sample.pulse_width_us) * 0.17F;
  sample.status = ClassifyDistance(sample.raw_distance_mm, min_distance_mm,
                                   max_distance_mm);
  if (sample.status == Status::VALID)
  {
    state.filtered_distance_mm = state.has_valid_sample
        ? state.filtered_distance_mm * filter_old_weight +
              sample.raw_distance_mm * (1.0F - filter_old_weight)
        : sample.raw_distance_mm;
    state.has_valid_sample = true;
  }
  sample.filtered_distance_mm = state.filtered_distance_mm;
  return sample;
}
inline Sample BuildNoEcho(const FilterState& state)
{
  Sample sample;
  sample.status = Status::NO_ECHO;
  sample.filtered_distance_mm = state.filtered_distance_mm;
  return sample;
}
}  // namespace Module::HC_SR04Logic
```

- [x] **Step 2: 在实现文件中完成构造与状态机**

`sr04.cpp` 使用以下核心实现，所有 `MSPM0PWM` 返回值都必须断言为 `OK`：

```cpp
#include "sr04.h"
#include "libxr_def.hpp"
#include "thread.hpp"
HC_SR04::HC_SR04(LibXR::ApplicationManager& app, const Resources& resources)
    : HC_SR04(app, resources, Config{})
{
}
HC_SR04::HC_SR04(LibXR::ApplicationManager& app, const Resources& resources,
                 const Config& config)
    : resources_(resources), config_(config), trigger_pwm_(resources.trigger_pwm),
      sample_topic_(LibXR::Topic::CreateTopic<Sample>(config.topic_name))
{
  ASSERT(resources_.capture_timer != nullptr);
  ASSERT(resources_.capture_clock_hz > 0U);
  ASSERT(resources_.capture_event_mask != 0U);
  ASSERT(config_.topic_name != nullptr);
  ASSERT(config_.trigger_frequency_hz > 0U);
  ASSERT(config_.trigger_pulse_width_us > 0U);
  ASSERT(config_.min_distance_mm <= config_.max_distance_mm);
  ASSERT(config_.filter_old_weight >= 0.0F && config_.filter_old_weight <= 1.0F);
  ASSERT(config_.no_echo_timeout_ms > 0U);
  const uint64_t high_us_per_second =
      static_cast<uint64_t>(config_.trigger_frequency_hz) *
      config_.trigger_pulse_width_us;
  ASSERT(high_us_per_second <= 1000000ULL);
  trigger_period_ms_ =
      (1000U + config_.trigger_frequency_hz - 1U) /
      config_.trigger_frequency_hz;
  ASSERT(trigger_pwm_.Disable() == LibXR::ErrorCode::OK);
  ASSERT(trigger_pwm_.SetConfig({config_.trigger_frequency_hz}) ==
         LibXR::ErrorCode::OK);
  const float duty_cycle = static_cast<float>(high_us_per_second) / 1000000.0F;
  ASSERT(trigger_pwm_.SetDutyCycle(duty_cycle) == LibXR::ErrorCode::OK);
  DL_Timer_clearInterruptStatus(resources_.capture_timer,
                                resources_.capture_event_mask);
  DL_Timer_startCounter(resources_.capture_timer);
  ASSERT(trigger_pwm_.Enable() == LibXR::ErrorCode::OK);
  last_capture_ms_ = LibXR::Thread::GetTime();
  last_publish_ms_ = last_capture_ms_;
  app.Register(*this);
}
void HC_SR04::Publish(Sample sample, uint32_t now_ms)
{
  sample.sequence = sequence_++;
  latest_sample_ = sample;
  last_publish_ms_ = now_ms;
  sample_topic_.Publish(latest_sample_);
}
void HC_SR04::OnMonitor()
{
  const uint32_t now_ms = LibXR::Thread::GetTime();
  const uint32_t event = DL_Timer_getRawInterruptStatus(
      resources_.capture_timer, resources_.capture_event_mask);
  if (event != 0U)
  {
    const uint32_t ticks = DL_Timer_getCaptureCompareValue(
        resources_.capture_timer, resources_.capture_channel);
    DL_Timer_clearInterruptStatus(resources_.capture_timer,
                                  resources_.capture_event_mask);
    last_capture_ms_ = now_ms;
    Publish(Module::HC_SR04Logic::ProcessCapture(
                ticks, resources_.capture_clock_hz, config_.min_distance_mm,
                config_.max_distance_mm, config_.filter_old_weight, filter_state_),
            now_ms);
    return;
  }
  if (static_cast<uint32_t>(now_ms - last_capture_ms_) <
          config_.no_echo_timeout_ms ||
      static_cast<uint32_t>(now_ms - last_publish_ms_) < trigger_period_ms_)
  {
    return;
  }
  Publish(Module::HC_SR04Logic::BuildNoEcho(filter_state_), now_ms);
}
LibXR::Topic& HC_SR04::SampleTopic() { return sample_topic_; }
const HC_SR04::Sample& HC_SR04::GetLatestSample() const { return latest_sample_; }
bool HC_SR04::HasValidSample() const { return filter_state_.has_valid_sample; }
```
